// middleware
    => function has access to req and res objects, يتم تنفيذها بعد استلام الريكويست وقبل ارسال الريسبوند
    => a custom logic, maybe for authentication or something else has a relationship with req or res 


// programm

const exp = require('express');
const app = exp();

app.use((req, res, next)=>{ // middleware
    console.log(`req method => ${req.method}, req url ${req.url}`);
    next(); // after execution go to next
})

app.get('/', (req, res)=>{
    res.send(`My first express program`);
})

app.get('/about', (req, res)=>{
    res.send(`About page is heree!`);
})

app.listen('5000', ()=>{
    console.log(`listening on port 5000`);
})
